# **试题B: 召唤数学精灵**

```python

def solve():
    target = 2024041331404202
    cycle_len = 200
    
    # 1. 计算第一个周期 (1-200) 中符合条件的个数
    # 此时需要考虑 B(i) 的实际值
    count_first_cycle = 0
    factorial_mod = 1
    for i in range(1, cycle_len + 1):
        # 计算 A(i) % 100
        a_i = (i * (i + 1) // 2) % 100
        
        # 计算 B(i) % 100
        if i < 10:
            factorial_mod = (factorial_mod * i) % 100
        else:
            factorial_mod = 0
            
        if a_i == factorial_mod:
            count_first_cycle += 1

    # 2. 计算后续标准周期 (如 201-400) 中符合条件的个数
    # 此时 i > 10, 所以 B(i) 恒为 0，只需判断 A(i) % 100 == 0
    count_standard_cycle = 0
    for i in range(1, cycle_len + 1):
        # 这里的 i 代表在一个周期内的偏移量
        if (i * (i + 1) // 2) % 100 == 0:
            count_standard_cycle += 1

    # 3. 计算总共有多少个完整的周期，以及余数
    num_full_cycles = target // cycle_len
    remainder = target % cycle_len

    # 总数 = 第一个周期 + (剩余完整周期数 * 标准周期数量)
    total = count_first_cycle + (num_full_cycles - 1) * count_standard_cycle

    # 4. 处理最后不足一个周期的部分
    # 这部分 i 肯定远大于 10，所以只需判断 A(i) % 100 == 0
    for i in range(1, remainder + 1):
        if (i * (i + 1) // 2) % 100 == 0:
            total += 1

    return total

print(solve())
```
