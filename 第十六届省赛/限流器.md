# 限流器

```python
import sys  
  
  
def solve():  
    # --- 调试与数据准备阶段 ---  
    # 在实际提交（OJ）时，应使用  
    raw_data = sys.stdin.read().split()  
    # sys.stdin.read() 会读取输入流中的所有内容  
    # .split() 会自动处理掉空格、制表符、换行符，将它们切分成一个个独立的字符串  
    # test_input = """  
    # 60 5 10    
    # 0 60 15 60 0 50 60 1 1 61    
    # """    
    #    # # 模拟从控制台获取数据并切分成列表  
    # raw_data = test_input.split()  
  
    # 如果列表为空，说明没有数据输入，直接退出防止报错  
    if not raw_data:  
        return  
  
    try:  
        # 将切分出来的字符串列表全量转换为整数列表  
        # 例如: ["60", "5"] -> [60, 5]  
        data = list(map(int, raw_data))  
    except ValueError:  
        # 如果数据中包含无法转为整数的内容，捕获异常并退出  
        return  
  
    # --- 参数解析阶段 ---  
    # 根据题目定义的格式提取前三个核心参数  
    n = data[0]  # 时间窗口的大小 N (例如 60 表示每 60 秒为一个周期)  
    m = data[1]  # 接口访问限制次数 M (每个周期内最多允许成功的次数)  
    l = data[2]  # 总请求次数 L (接下来的数据量)  
  
    # 从索引 3 开始，截取长度为 l 的子列表作为所有的访问时间戳  
    # 这样可以确保即使输入数据后面有多余的数字，也不会影响统计结果  
    timestamps = data[3: 3 + l]   # timestamps 所有的访问时间戳  
  
    # --- 核心限流统计逻辑 ---  
    # window_counts 用于记录每一个具体时间窗内“已经成功”的次数  
    # Key: 窗口的编号 (k), Value: 该窗口内当前成功的计数  
    window_counts = {}  
  
    # 用于累计所有时间窗内成功的总次数  
    total_success = 0  
  
    # 遍历每一个请求的时间戳  
    for t in timestamps:  
        # 【关键算法】：计算当前时间戳 t 属于哪一个窗口 ID        
        # 使用整除符号 //。例如：n=60 时  
        # t=0...59 结果均为 0 (属于第 0 号窗口)  
        # t=60...119 结果均为 1 (属于第 1 号窗口)  
        k = t // n  
  
        # 从字典中获取当前窗口 ID 已经成功的次数，如果还没记录过则默认为 0        
        current_count = window_counts.get(k, 0)  
  
        # 判断是否达到限流阈值  
        if current_count < m:  
            # 如果没满，则该请求成功  
            # 1. 更新该窗口的成功计数  
            window_counts[k] = current_count + 1  
            # 2. 全局成功总数加 1            total_success += 1  
  
            # 调试日志：print(f"时间 {t} 在窗口 {k}：成功 (第 {window_counts[k]} 次)")  
        else:  
            # 如果已达 M 次，则该请求失败，直接跳过（不进入统计）  
            pass  
            # 调试日志：print(f"时间 {t} 在窗口 {k}：失败 (已达上限 {m})")  
    # --- 结果输出 ---    # 打印最终满足条件的成功请求总数  
    print(total_success)  
  
  
if __name__ == "__main__":  
    solve()
```
