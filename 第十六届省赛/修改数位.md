# 修改数位

```python
import sys  
  
  
def solve():  
    # 模拟输入：123456789301  
    # s_str = "123456789301"  # 没有前导零  
    s_str=sys.stdin.readline().strip()  
    n = len(s_str)  
    # 题目要求存在连续的 10 位，如果原数长度不足 10，直接返回  
    if n < 10:  
        return  
  
    # 将字符串拆解为整数列表，方便后续进行减法运算  
    # 例如 "123" -> [1, 2, 3]    digits = [int(d) for d in s_str]  
  
    # 初始化最小花费为正无穷大，用于后续对比取最小值  
    min_total_cost = float('inf')  
  
    # --- 核心逻辑：滑动窗口 ---    # 遍历所有可能的起始位置 i。由于窗口长度是 10，所以 i 的范围是 0 到 n-10    for i in range(n - 9):  
        # 截取当前的 10 位数字作为一个窗口  
        window = digits[i:i + 10]  # window 当前窗口  
  
        # 逻辑判断：是否涉及“前导零”约束  
        # 修改后的数不能有前导零，意味着如果窗口在原数的最开头（i=0），  
        # 那么窗口的第一个数字（即原数的最高位）修改后不能为 0。  
  
        if i > 0:  # i当前窗口在原数字中的起始索引（位置）  
            # 【情况 1：普通窗口（不在开头）】  
            # 贪心策略：要使代价和最小，只需将窗口内的 10 个数从小到大排序，  
            # 然后分别与目标数字 [0, 1, 2, ..., 9] 一一对应相减取绝对值。  
            sorted_win = sorted(window)  # sorted_win 排列好的数据  
            current_cost = sum(abs(sorted_win[j] - j) for j in range(10))  
  
        else:  
            # 【情况 2：首位窗口（在开头）】  
            # 因为最高位不能为 0，所以我们需要枚举最高位可能变成的数字 k (1-9)            current_cost = float('inf')  # 正无穷大  
            first_digit = window[0]  # 窗口的第一位（也是原数的最高位）  
            rest_digits = sorted(window[1:])  # 窗口剩下的 9 位，先排好序  
  
            # 枚举最高位变成 k (k 取值 1 到 9)            for k in range(1, 10):  
                # 如果最高位选了 k，那么剩下的 9 位就要去匹配 [0-9] 中除去 k 以外的数字  
                targets = [t for t in range(10) if t != k]  
  
                # 计算当前枚举下的花费：  
                # 1. 最高位的修改代价  
                cost_k = abs(first_digit - k)  
                # 2. 剩余 9 位按顺序匹配的代价总和  
                for j in range(9):  
                    cost_k += abs(rest_digits[j] - targets[j])  
  
                # 在所有 k (1-9) 的尝试中，保留代价最小的那一个  
                if cost_k < current_cost:  
                    current_cost = cost_k  
  
        # --- 更新全局最优解 ---        # 如果当前窗口计算出的代价比之前记录得更小，则更新它  
        if current_cost < min_total_cost:  
            min_total_cost = current_cost  
  
    # 输出遍历完所有窗口后的最小总代价  
    print(min_total_cost)  
  
  
if __name__ == "__main__":  
    solve()
```
