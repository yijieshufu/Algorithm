```python
import sys
def solve():

    input_data = sys.stdin.read().split()

    if not input_data:

        return

    n = int(input_data[0])

    a = list(map(int, input_data[1:]))

    # 所有的原始间隔

    gaps = []

    for i in range(n - 1):

        gaps.append(a[i+1] - a[i])

    ans = 0

    # 当前能够使用的“最大起始步长”

    # 第一段间隔 gaps[0] 之前没有参考，所以第一段本身不需要插入

    # 我们从第二段开始，它的第一个小间距受限于第一段（或第一段插入后的最后一段）

    cur_pre_gap = gaps[0]

    for i in range(1, n - 1):

        target = gaps[i]

        # 贪心：如果 target 超过了前一个间距的两倍，必须插入

        # 这里需要注意的是，我们要让 target 在插入后，最后一段留在结尾的间距尽量大

        # 这样对下一个原始间隔最有利。

        while target > 2 * cur_pre_gap:

            ans += 1

            cur_pre_gap = cur_pre_gap * 2

            target -= cur_pre_gap

        # 重要：这一段处理完后，最后留下的那一小段就是下一段的参考 pre_gap

        cur_pre_gap = target

  

    print(ans)

  

if __name__ == "__main__":

    solve()
```
