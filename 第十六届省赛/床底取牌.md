```python
import sys  
def get_password_strength(s):  
    # 1. 定义合法字符集  
    upper = "ABCDEFGHIJKLMNOPQRSTUVWXYZ"  
    lower = "abcdefghijklmnopqrstuvwxyz"  
    digits = "0123456789"  
    specials = "~!@#$%^&*()_"  
    allowed_chars = set(upper + lower + digits + specials)  # allowed_chars 构建一个set集合里面全是上面的叠加的元素  
  
    # 2. 检查是否有非法字符  
    for char in s:  
        if char not in allowed_chars:  
            return 0  
  
    length = len(s)  
    # 记录包含的种类  
    # any 只要括号里的序列中有一个值是 True    has_upper = any(c in upper for c in s)  
    has_lower = any(c in lower for c in s)  
    has_digit = any(c in digits for c in s)  
  
  
    # 统计特殊字符种类数  
    found_specials = set()  # found_specials 判断一下里面有几个特殊字符  
    for char in s:  
        if char in specials:  
            found_specials.add(char)  
  
    num_specials = len(found_specials)  # num_specials 特殊字符的长度  
    has_special = num_specials > 0  # has_special 标记一下是否有特殊字符  
  
    # 计算总类别数（大写、小写、数字、特殊字符）  
    categories_count = sum([has_upper, has_lower, has_digit, has_special])  
  
    # 3. 强度判断逻辑 (按从强到弱判断)  
  
    # 强密码 (3): 长度 >= 12 且 (4种全有 OR (3种且特殊字符 >= 3))    is_strong = (length >= 12) and (  
            categories_count == 4 or (categories_count == 3 and num_specials >= 3)  
    )  
    if is_strong:  
        return 3  
  
    # 中密码 (2): 长度 >= 8 且 至少包含 2 种  
    if length >= 8 and categories_count >= 2:  
        return 2  
  
    # 弱密码 (1): 长度 >= 6    if length >= 6:  
        return 1  
  
    # 不合法 (0): 其他所有情况  
    return 0  
  
  
def solve():  
    # 读取所有输入，处理大数据量时 sys.stdin 效率更高  
    input_data = sys.stdin.read().splitlines()  
  
    # # 手动构造测试数据：第一行是个数，后面是密码  
    # input_data = ["1", "@Qaq12345"]  
  
    if not input_data:  
        return  
  
  
    t = int(input_data[0]) # t 判断密码的个数  
    results = []  
    for i in range(1, t + 1): # i=1  
        results.append(str(get_password_strength(input_data[i])))  
  
    # 批量输出  
    sys.stdout.write("\n".join(results) + "\n")  
  
  
if __name__ == "__main__":  
    solve()
```
