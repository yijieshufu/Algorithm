# 商品库存管理

**频繁对某个区间进行加减操作，最后才询问结果”** 的题目，脑子里第一时间就要蹦出 **“差分数组**

```python
import sys

def solve():
    # --- 1. 快速读入阶段 ---
    # 使用 sys.stdin.read().split() 避免多次磁盘 I/O，适合处理 10^5 级别的数据
    input_data = sys.stdin.read().split()
    if not input_data:
        return
    
    n = int(input_data[0])  # 商品种类总数
    m = int(input_data[1])  # 计划执行的操作总数
    
    # 记录所有操作的原始区间，因为最后要逐个“假设”撤销每一个操作 
    # 一共是有m次操作
    ops = []
    idx = 2
    for _ in range(m):
        l = int(input_data[idx])
        r = int(input_data[idx+1])
        ops.append((l, r))
        idx += 2
        
    # --- 2. 差分数组预处理 ---
    # diff[i] 记录 i 位置相对于 i-1 位置的增量。
    # 这样区间 [L, R] 加 1，只需修改两个点：diff[L] 和 diff[R+1]
    # 差分数组通过在 $L$ 点‘开启’增量并在 $R+1$ 点‘关掉’抵消，巧妙利用前缀和的传递性，将繁琐的区间覆盖简化成了极速的点对点标记
    diff = [0] * (n + 2)
    for l, r in ops:
        diff[l] += 1      # 区间起点：库存开始增加
        diff[r + 1] -= 1  # 区间终点的下一位：库存增加结束
        
    # counts[i] 存储执行完所有 m 个操作后，商品 i 的最终库存量
    counts = [0] * (n + 1) # 最终库存量
    current_stock = 0     # 累加器，用来还原差分数组
    total_zeros = 0       # 统计：如果所有操作都做，原本就有多少商品库存为 0
    
    # 还原数组：通过累加差分值，得到每个位置的真实库存
    for i in range(1, n + 1):
        current_stock += diff[i] 
        # 如果说差分标记是写下的“入场”和“离场”指令，那么这一步前缀和就是**正式按照清单点名，算出每一个时刻场内到底有多少人**
        counts[i] = current_stock
        if counts[i] == 0:
            total_zeros += 1
            
    # --- 3. 核心逻辑：分析撤销贡献 ---
    # 关键思考：如果不执行某个操作 [L, R]，哪些商品会变成 0？
    # 只有那些“原本库存正好为 1 且在 [L, R] 范围内”的商品会变成 0。
    # 我们用前缀和数组 p1 记录：到位置 i 为止，有多少个商品的库存恰好是 1。
    p1 = [0] * (n + 1)
    for i in range(1, n + 1):
        # 这是一个计数前缀和：当前是1则+1，否则保持不变
        p1[i] = p1[i-1] + (1 if counts[i] == 1 else 0)
        
    # --- 4. 计算并输出结果 ---
    results = []
    for l, r in ops:
        # 对于每一个操作 [L, R]，如果撤销它：
        # 最终 0 的数量 = (原本就是 0 的) + (区间内因为减1而变成 0 的)
        # 使用 p1[r] - p1[l-1] 可以 O(1) 算出区间内库存为 1 的商品数
        ones_in_range = p1[r] - p1[l-1]
        results.append(str(total_zeros + ones_in_range))
        
    # 一次性输出，提升效率
    sys.stdout.write("\n".join(results) + "\n")

if __name__ == "__main__":
    solve()
```
