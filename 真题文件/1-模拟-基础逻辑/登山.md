## 题目 3339: 登山

```python
import sys

# 提高递归深度以防万一，但我们会使用迭代版 find
sys.setrecursionlimit(10**6 + 100)

def solve():
    # 1. 快速读取数据
    input_data = sys.stdin.read().split()
    if not input_data:
        return
    
    n = int(input_data[0])
    m = int(input_data[1])
    
    # 扁平化存储矩阵，节省内存和索引开销
    grid = []
    for i in range(2, len(input_data)):
        grid.append(int(input_data[i]))
    
    # 2. 并查集 (DSU) 工具函数
    parent = list(range(n * m))
    
    def find(i):
        """路径压缩的迭代版 find，防止在大规模数据下栈溢出"""
        root = i
        while parent[root] != root:
            root = parent[root]
        while parent[i] != root:
            next_node = parent[i]
            parent[i] = root
            i = next_node
        return root

    def union(i, j):
        """合并两个集合"""
        root_i = find(i)
        root_j = find(j)
        if root_i != root_j:
            parent[root_i] = root_j

    # 3. 处理行方向的连通性
    for r in range(n):
        row_start = r * m  # 计算当前行在扁平化数组 grid 中的起始位置
        # 预计算前缀最大值
        p_max = [0] * m #`p_max` 生成一个全是0的列表
        p_max[0] = grid[row_start]  # 初始化第一个元素
        for c in range(1, m): 
	        # p_max[c]：我们要填入的目标，即“从开头到当前位置 c 的最大值”
	        # p_max[c-1]：我们已经算好的结果，即“从开头到前一个位置 c-1 的最大值”
	        # grid[row_start + c]：当前这一步新看到的格子高度
            p_max[c] = max(p_max[c-1], grid[row_start + c])
        
        # 预计算后缀最小值
        s_min = [0] * m
        s_min[m-1] = grid[row_start + m - 1]
        for c in range(m-2, -1, -1):
            s_min[c] = min(s_min[c+1], grid[row_start + c])
        
        # 检查每个相邻位置是否可以“打通”
        for c in range(m - 1):
            if p_max[c] > s_min[c+1]:
                union(row_start + c, row_start + c + 1)

    # 4. 处理列方向的连通性
    for c in range(m):
        # 预计算列的前缀最大和后缀最小
        p_max = [0] * n
        p_max[0] = grid[c]
        for r in range(1, n):
            p_max[r] = max(p_max[r-1], grid[r * m + c])
            
        s_min = [0] * n
        s_min[n-1] = grid[(n-1) * m + c]
        for r in range(n-2, -1, -1): 
	        # 完整格式是 `range(start, stop, step)`
	        # `-1`。循环到这个值之前停止（不包括 -1）也就是到0
	        # -1会递减
            s_min[r] = min(s_min[r+1], grid[r * m + c])
            
        for r in range(n - 1):
            if p_max[r] > s_min[r+1]:
                union(r * m + c, (r + 1) * m + c)

    # 5. 计算结果
    # 第一步：找出每个连通块的最大值
    # 初始化一个大小为 n*m 的数组，用来存放每个根节点（老大）对应的最高高度
    max_heights = [0] * (n * m) 
    for i in range(n * m):
	    # 找到当前格子 i 所属帮派的“老大”（根节点）
        root = find(i)
        # 看看当前格子 i 的高度，是否刷新了该帮派的历史纪录
        if grid[i] > max_heights[root]:
            max_heights[root] = grid[i]
            
    # 第二步：累加每个格子能达到的最高高度
    total_sum = 0
    for i in range(n * m):
	    # 再次遍历每个格子，找到它所属帮派的老大
	    # 并通过老大拿到该帮派的最高高度
        total_sum += max_heights[find(i)]
        
    # 输出平均值，保留6位小数
    #f"..."`: 字符串前面的 `f` 告诉 Python，这个字符串里面包含需要被计算的表达式。
	#{...}: 花括号内部是我们要计算的内容。
	#`:`   分隔符。冒号左边是计算结果，右边是格式说明。 
    #`.6` 精度说明符。它告诉 Python：“请在这个数字的小数点后保留 **6位**。    
    # `f`   类型说明符。代表这是一个**浮点数（float）**。
    print(f"{total_sum / (n * m):.6f}")

if __name__ == "__main__":
    solve()
```
