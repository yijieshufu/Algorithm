class Solution:

    def countPrimes(self, n: int) -> int:
        # 如果 n 小于 3，则没有小于 n 的质数（2 是最小的质数）
        if n < 3:
            return 0
        # 初始化布尔列表，默认全是质数 (True)
        # 列表长度为 n，索引从 0 到 n-1
        primes = [True] * n
        primes[0] = primes[1] = False # 0 和 1 不是质数
        # 埃氏筛优化：只需要遍历到 sqrt(n)
        for i in range(2, int(n**0.5) + 1):
            if primes[i]:
                # 列表切片优化：
                # 从 i*i 开始筛选（因为 2i, 3i 等已经在之前的循环中被筛掉了）
                # 步长为 i，将所有倍数标记为 False
                # len(primes[i*i:n:i]) 计算需要赋值的长度
                primes[i*i : n : i] = [False] * len(range(i*i, n, i))
        # 返回 True 的总数
        return sum(primes)
