## 题目 3338:  最大数字

```python
import sys
from functools import cmp_to_key

# 关键：解除 Python 默认的整数转字符串 4300 位的长度限制
# 研究生组的题目 $n$ 很大时，结果往往会突破这个限制
sys.set_int_max_str_digits(0)

def solve():
    # 使用快速读入
    line = sys.stdin.read().strip()
    if not line:
        return
    n = int(line)

    # 1. 生成二进制字符串列表
    # bin(i)[2:] 去掉 '0b' 前缀
    bin_list = [bin(i)[2:] for i in range(1, n + 1)]

    # 2. 核心贪心逻辑：自定义排序
    # 对于两个二进制串 a 和 b，比较 a+b 和 b+a 谁组成的数字更大
    def compare(a, b):
        if a + b > b + a:
            return -1  # a 应该排在前面
        elif a + b < b + a:
            return 1   # b 应该排在前面
        else:
            return 0

    # 排序：时间复杂度 O(N log N * L)，L 为平均字符串长度
    bin_list.sort(key=cmp_to_key(compare))

    # 3. 拼接并输出
    # int(s, 2) 将二进制串转为大整数，支持无限精度
    max_bin_str = "".join(bin_list)
    print(int(max_bin_str, 2))

if __name__ == "__main__":
    solve()
```
