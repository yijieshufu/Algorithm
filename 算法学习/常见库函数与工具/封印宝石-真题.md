# 题目描述
- **题目痛点**：我们要填满 $n$ 个盒子，每个盒子都要选一个魔力值。规则有两个：
    1. **体力限制**：把第 $i$ 颗宝石放入第 $j$ 个盒子（$j \le i$）消耗 $i-j$ 点体力。
    2. **字典序最大**：我们要让前面的盒子放魔力值尽量大的宝石，且相邻盒子魔力值不能相同。
# 代码

```python
import sys
it = iter(sys.stdin.read().split())
n, k = int(next(it)), int(next(it))
a = [int(next(it)) for _ in range(n)]
# 2. 迭代线段树初始化 (sz 为叶子数)
sz = 1
while sz <n: sz<<=1
# t 存储 [魔力值, 原始下标]
t = [[-1,-1] for _ in range(2*sz)]
def upd(i,v):
  p =i+sz-1 #最低端
  t[p]=[v,i] # 赋值
  while p>1: #使得t中第一个是最强
    p//=2 # 找父节点
    if t[2*p][0] >= t[2*p+1][0]:t[p]=t[2*p]
    else:t[p]=t[2*p+1]
def qry(l,r):
  l,r=l+sz-1,r+sz-1 #从 底部开始
  res=[-1,-1] #记录最强的
  while l<=r:
    if l%2==1:
      if t[l][0]>res[0]:res=t[l]
      l+=1
    if r%2==0:
      if t[r][0]>res[0]:res=t[r]
      r-=1
    l//=2;r//=2 # 往上走
  return res
# 填入宝石
for i in range(n):upd(i+1,a[i])
ans = [] # 盒子里 放入的宝石
lst = -1
for j in range(1,n+1):# j 盒子的编号
  lim = min(n,j+k) #  当前最远够的着的盒子
  best =qry(j,lim) #
  if best[0]!= -1 and best[0]==lst: # 找到了最强的 但是重复了
    mx_v,mx_idx=best
    upd(mx_idx,-1) #冠军暂离
    best = qry(j,lim) # 拿到亚军
    upd(mx_idx,mx_v) # 冠军归位
  if best[0]>0: #没有重复哦 直接放
    v,idx=best
    ans.append(v)
    k-=(idx-j) #消耗k值 宝石的下标idx 大于 盒子下标的
    lst = v 
    upd(idx,-2) # 该宝石永久离场
  else:
    ans.append(-1)
    lst=-1
print(*ans)
```
