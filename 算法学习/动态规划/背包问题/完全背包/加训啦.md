
**二分答案**  +    **完全背包 DP** 的综合题。
### 代码实现与详细注解
```Python
import sys
from collections import Counter

# 1. 指针式快速读入
d = list(map(int, sys.stdin.read().split()))
p = 0

n = d[p]; p += 1
m = d[p]; p += 1
k = d[p]; p += 1

# 统计每个力量值有多少士兵，优化大规模 n 的遍历
a_list = d[p : p + n]; p += n
b = d[p : p + m]; p += m
c = d[p : p + m]; p += m

cnt = Counter(a_list)
# 排序后的不同力量值及其人数：(力量, 人数)
items = sorted(cnt.items())
low_a = items[0][0]

# 2. 预处理完全背包
# M 是根据题目精力上限与提升效率估算的合理增量范围
M = 20005
inf = k + 7
f = [0] + [inf] * M

for i in range(m):
    gn, ct = b[i], c[i]
    if ct > k: continue
    # 标准完全背包：每个计划可用无限次
    for j in range(gn, M + 1):
        if f[j - gn] + ct < f[j]:
            f[j] = f[j - gn] + ct

# 单调化处理：使得 f[j] 表示提升“至少” j 力量的代价
for j in range(M - 1, -1, -1):
    if f[j + 1] < f[j]:
        f[j] = f[j + 1]

# 3. 二分判定函数
def check(x):
    tot = 0
    for v, num in items:
        if v >= x: break # 当前士兵已达标，后续更强的也一定达标
        diff = x - v
        if diff > M: return False # 增量需求超出预处理范围
        tot += f[diff] * num
        if tot > k: return False # 及时止损
    return True

# 4. 执行二分
l, r = low_a, low_a + M - 1
ans = low_a
while l <= r:
    mid = (l + r) // 2
    if check(mid):
        ans = mid
        l = mid + 1
    else:
        r = mid - 1

print(ans)
```
