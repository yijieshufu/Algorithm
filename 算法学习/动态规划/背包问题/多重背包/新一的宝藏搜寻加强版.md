
### Python 代码实现
```Python
import sys
from collections import deque

# 提高读取效率
input = sys.stdin.readline
n, m = map(int, input().split())
f = [0] * (m + 1)

for _ in range(n):
    v, w, s = map(int, input().split())
    g = f[:]  # 备份上一轮状态，滚动数组优化
    for r in range(v):
        q = deque()
        for j in range((m - r) // v + 1):
            # 计算当前状态的对比值：f[t*v + r] - t*w
            val = g[j * v + r] - j * w
            # 维护单调递减队列
            while q and g[q[-1] * v + r] - q[-1] * w <= val:
                q.pop()
            q.append(j)
            # 移出滑动窗口（数量限制 s）
            if q[0] < j - s:
                q.popleft()
            # 更新 dp 状态
            f[j * v + r] = g[q[0] * v + r] - q[0] * w + j * w

print(f[m])
```
