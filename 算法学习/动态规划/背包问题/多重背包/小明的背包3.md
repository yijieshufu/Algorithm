

这是一道经典的**多重背包问题**（Multiple Knapsack Problem）。

### 题目分析

在多重背包中，每种物品 $i$ 有固定的体积 $w_i$、价值 $v_i$ 和限定的数量 $s_i$。我们的目标是在总体积不超过 $V$ 的情况下，最大化总价值。

# 二进制代码
```Python
import sys

# 1. 快速读入：n 是物品种类，m 是背包总容量
it = iter(sys.stdin.read().split())
n, m = int(next(it)), int(next(it))

# 2. 二进制拆分：将多重背包转换为 01 背包物品清单
qs = []
for _ in range(n):
    # 根据题目：wi 为体积，vi 为价值，si 为数量
    v, w, s = int(next(it)), int(next(it)), int(next(it))
    k = 1
    while k <= s:
        qs.append((k * v, k * w))
        s -= k
        k <<= 1
    if s > 0:
        qs.append((s * v, s * w))

# 3. 标准 01 背包状态转移
f = [0] * (m + 1)
for v, w in qs:
    # 逆序遍历，确保 01 背包逻辑
    for j in range(m, v - 1, -1):
        if f[j - v] + w > f[j]:
            f[j] = f[j - v] + w

print(f[m])
```

 
# 单调队列优化 代码
```Python
import sys
from collections import deque

# 快速 I/O
it=iter(sys.stdin.read().split())

def solve():
    n, m = int(next(it)), int(next(it))
    # f[j] 记录当前最大价值
    f = [0] * (m + 1)
    for _ in range(n):
        w, v, s = int(next(it)), int(next(it)), int(next(it))
        # 备份上一轮的 dp 状态，单调队列需要基于“上一层”的状态更新
        g = f[:]
    
        # 按余数 r 分组
        for r in range(w):
            q = deque()
            # 遍历每一个可能的商 k (j = k*w + r)
            for k in range((m - r) // w + 1):
                # 计算当前待入队的值：g[k*w + r] - k*v
                val = g[k * w + r] - k * v
                # 维护队列单调递减
                while q and g[q[-1] * w + r] - q[-1] * v <= val:
                    q.pop()
                q.append(k)
                # 检查队头是否过期（窗口大小为 s+1）
                if q[0] < k - s:
                    q.popleft()
                
                # 更新 dp 值：加上当前的 q*v
                f[k * w + r] = max(f[k * w + r], g[q[0] * w + r] - q[0] * v + k * v)

    print(f[m])

solve()
```
