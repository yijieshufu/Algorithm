```python
import sys
from collections import deque

# 读取输入
line = sys.stdin.readline()
if not line: exit()
n = int(line.strip())

# dist[i] 表示余数为 i 时，二进制中 1 的最小个数
dist = [float('inf')] * n
dq = deque()

# 初始状态：数字 1，余数为 1 % n，1 的个数为 1
start_node = 1 % n
dist[start_node] = 1
dq.append(start_node)

while dq:
    u = dq.popleft()
    
    # 操作 1: X * 2 (代价 +0) -> 放到队首
    v1 = (u * 2) % n
    if dist[v1] > dist[u]:
        dist[v1] = dist[u]
        dq.appendleft(v1)
        
    # 操作 2: X + 1 (代价 +1) -> 放到队尾
    v2 = (u + 1) % n
    if dist[v2] > dist[u] + 1:
        dist[v2] = dist[u] + 1
        dq.append(v2)

# 输出到达余数 0 的最小代价
print(dist[0])
```
