class Solution:  
    def smallestRepunitDivByK(self, k: int) -> int:  
        # 1. 特判：偶数或以 5 结尾的数永远无法整除只含 1 的数  
        if k % 2 == 0 or k % 5 == 0:  
            return -1  
        # 2. 初始化  
        dist = [-1] * k  
        start = 1 % k  # 初始数字是 1，其余数是 1 % k        
        dist[start] = 1  # 记录余数为 start 时，数字长度为 1        
        dq = collections.deque([start])  
        # 3. BFS 搜索  
        while dq:  
            u = dq.popleft()  
            # 找到终点：余数为 0            
            if u == 0:  
                return dist[u]  # 注意：这里必须用 return！  
            # 转移操作：在末尾加个 '1'，新数 = 旧数 * 10 + 1            
            v = (u * 10 + 1) % k  # v：0 -   v-1
            if dist[v] == -1:  
                dist[v] = dist[u] + 1  
                dq.append(v)  
        return -1
