## 题目 3330:01 串 后面再理解

```python
import sys

# 解决大数据量下整数转字符串的长度限制（如果涉及大规模打印输出）
sys.set_int_max_str_digits(0)

def get_total_len(n):
    """计算从 0 到 n 所有整数二进制拼接后的总长度"""
    if n < 0: return 0
    if n == 0: return 1
    
    # 0 占 1 位
    total_len = 1
    # 计算从 1 到 n 的部分
    # 对于 k 位二进制数（如 2 位数是 2~3），数量是 2^(k-1)
    k = 1
    while (1 << k) <= n:
        # 长度为 k 的二进制数范围是 [2^(k-1), 2^k - 1]
        count = (1 << (k - 1))
        total_len += count * k
        k += 1
    
    # 处理最后一组不足 2^k 的部分（即从 2^(k-1) 到 n）
    remaining_count = n - (1 << (k - 1)) + 1
    total_len += remaining_count * k
    return total_len

def count_ones_in_range(n):
    """统计从 0 到 n 所有整数二进制表示中 '1' 的总个数"""
    if n < 0: return 0
    count = 0
    i = 0
    while (1 << i) <= n:
        # 计算第 i 位（从低到高，0开始）上 1 出现的周期
        total_pairs = (n + 1) // (1 << (i + 1))
        count += total_pairs * (1 << i)
        
        # 处理非完整周期内的剩余部分
        remaining = (n + 1) % (1 << (i + 1))
        count += max(0, remaining - (1 << i))
        i += 1
    return count

def solve():
    # 读取 x
    line = sys.stdin.readline().strip()
    if not line: return
    x = int(line)
    
    # 1. 二分查找确定 x 落在哪个数字 N 的二进制表示中
    low = 0
    high = x  # N 不可能超过 x
    n = 0
    while low <= high:
        mid = (low + high) // 2
        if get_total_len(mid) >= x:
            n = mid
            high = mid - 1
        else:
            low = mid + 1
            
    # 2. 计算 0 到 N-1 中所有的 1 的个数
    ans = count_ones_in_range(n - 1)
    
    # 3. 处理数字 N 本身的前缀部分
    # 计算前面的总长度，确定我们需要从数字 N 中取前几位
    prev_len = get_total_len(n - 1)
    rem_len = x - prev_len
    
    # 获取 N 的二进制字符串（去掉 '0b'）
    bin_n = bin(n)[2:]
    # 统计前 rem_len 位中 1 的个数
    ans += bin_n[:rem_len].count('1')
    
    print(ans)

if __name__ == "__main__":
    solve()
```
