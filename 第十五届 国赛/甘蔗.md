## 题目3340 甘蔗

```java
import java.io.*;
import java.util.*;

public class Main {
    static final int INF = 0x3f3f3f3f;

    public static void main(String[] args) throws IOException {
        // 使用快速读入提升性能
        StreamTokenizer st = new StreamTokenizer(new BufferedReader(new InputStreamReader(System.in)));
        
        autoNextInt(st);
        int n = (int) st.nval;
        autoNextInt(st);
        int m = (int) st.nval;

        int[] a = new int[n];
        int maxA = 0;
        for (int i = 0; i < n; i++) {
            autoNextInt(st);
            a[i] = (int) st.nval;
            if (a[i] > maxA) maxA = a[i];
        }

        // 使用 Set 去重，避免重复计算相同的高度差值
        Set<Integer> bSetTemp = new HashSet<>();
        for (int i = 0; i < m; i++) {
            autoNextInt(st);
            bSetTemp.add((int) st.nval);
        }
        Integer[] bSet = bSetTemp.toArray(new Integer[0]);

        // dp[j] 表示当前位置高度为 j 时的最小砍伐次数
        int[] dp = new int[maxA + 1];
        Arrays.fill(dp, INF);

        // 初始化第一根甘蔗
        for (int j = 0; j <= a[0]; j++) {
            dp[j] = (j == a[0] ? 0 : 1);
        }

        // 动态规划转移
        for (int i = 1; i < n; i++) {
            int[] nextDp = new int[maxA + 1];
            Arrays.fill(nextDp, INF);

            // 预先获取前一根甘蔗可能的高度上限
            int prevLimit = a[i - 1];
            int currentLimit = a[i];

            for (int j = 0; j <= currentLimit; j++) {
                int minPrev = INF;
                // 遍历差值集合 B
                for (int b : bSet) {
                    // 满足 |j - k| = b => k = j + b 或 k = j - b
                    int k1 = j - b;
                    if (k1 >= 0 && k1 <= prevLimit) {
                        if (dp[k1] < minPrev) minPrev = dp[k1];
                    }
                    int k2 = j + b;
                    if (k2 <= prevLimit) {
                        if (dp[k2] < minPrev) minPrev = dp[k2];
                    }
                }

                if (minPrev != INF) {
                    nextDp[j] = minPrev + (j == currentLimit ? 0 : 1);
                }
            }
            dp = nextDp; // 滚动更新
        }

        // 获取最后一根的所有可能高度中的最小值
        int result = INF;
        for (int j = 0; j <= a[n - 1]; j++) {
            if (dp[j] < result) result = dp[j];
        }

        System.out.println(result >= INF ? -1 : result);
    }

    private static void autoNextInt(StreamTokenizer st) throws IOException {
        st.nextToken();
    }
}
```
