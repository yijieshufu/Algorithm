```Python
    if b == 0:
        return a, 1, 0
    d, x_prev, y_prev = exgcd(b, a % b)
    x = y_prev
    y = x_prev - (a // b) * y_prev
```

```Python
def mul(a, b, n):
    c = [[0] * (n + 1) for _ in range(n + 1)]
    for i in range(1, n + 1):
        for k in range(1, n + 1):
            if not a[i][k]: continue
            for j in range(1, n + 1):
                c[i][j] = (c[i][j] + a[i][k] * b[k][j]) % mod
    return c

def qpow_matrix(a, k, n):
    res = [[0] * (n + 1) for _ in range(n + 1)]
    for i in range(1, n + 1): res[i][i] = 1
    while k:
        if k & 1: res = mul(res, a, n)
        a = mul(a, a, n)
        k >>= 1
    return res
```

```Python
# 约定：L 为大根堆（存较小一半，取负值），R 为小根堆（存较大一半）

def add_to_dual_heap(val, L, R):
```

```Python
def binary_search(l, r, check_func):
    ans = l
    while l <= r:
        mid = (l + r) // 2
        if check_func(mid):
            ans = mid
            l = mid + 1  # 满足条件，尝试寻找更大的可能
```

##  Dijkstra 单源最短路 (纯函数化)


```Python
import heapq

def dijkstra(s, n, g):
    dist = [float('inf')] * (n + 1)
    dist[s] = 0
    pq = [(0, s)]
    while pq:
        d, u = heapq.heappop(pq)
        if d > dist[u]: continue
        for v, w in g[u]:
            if dist[u] + w < dist[v]:
                dist[v] = dist[u] + w
                heapq.heappush(pq, (dist[v], v))
    return dist
```

## 树链剖分 (HLD)
```Python
# 【模板积木 1】树链剖分 (HLD)
# 功能：建立重链剖分，维护 dfs 序、深度、父节点、重儿子、链顶
def hld_init(n, g, root):
    sz = [0] * (n + 1)
    dep = [0] * (n + 1)
    fa = [0] * (n + 1)
    son = [0] * (n + 1)
    
    stack = [(root, 0, 1)]
    order = []
    while stack:
        u, p, d = stack.pop()
        fa[u] = p
        dep[u] = d
        sz[u] = 1
        order.append(u)
        for v in g[u]:
            if v != p:
                stack.append((v, u, d + 1))
    
    for u in reversed(order):
        if fa[u]:
            sz[fa[u]] += sz[u]
            if son[fa[u]] == 0 or sz[u] > sz[son[fa[u]]]:
                son[fa[u]] = u

    top = [0] * (n + 1)
    dfn = [0] * (n + 1)
    rnk = [0] * (n + 1)
    cur_dfn = 0
    
    stack = [(root, root)]
    while stack:
        u, t = stack.pop()
        cur_dfn += 1
        dfn[u] = cur_dfn
        rnk[cur_dfn] = u
        top[u] = t
        
        # 先压轻儿子，再压重儿子，保证重儿子先出栈（DFN 连续）
        for v in g[u]:
            if v != fa[u] and v != son[u]:
                stack.append((v, v))
        if son[u]:
            stack.append((son[u], t))
            
    return dfn, rnk, top, fa, dep, sz
```
## 线段树 (Lazy Tag)
```Python
# 【模板积木 2】线段树 (Lazy Tag)
# 功能：区间加、区间求和
def build(node, l, r, rnk, a, tree):
    if l == r:
        tree[node] = a[rnk[l]]
        return
    mid = (l + r) // 2
    build(node * 2, l, mid, rnk, a, tree)
    build(node * 2 + 1, mid + 1, r, rnk, a, tree)
    tree[node] = tree[node * 2] + tree[node * 2 + 1]

def push_down(node, l, r, tree, lazy):
    if lazy[node] != 0:
        mid = (l + r) // 2
        lazy[node * 2] += lazy[node]
        tree[node * 2] += lazy[node] * (mid - l + 1)
        lazy[node * 2 + 1] += lazy[node]
        tree[node * 2 + 1] += lazy[node] * (r - mid)
        lazy[node] = 0

def update(node, l, r, ql, qr, val, tree, lazy):
    if ql <= l and r <= qr:
        tree[node] += val * (r - l + 1)
        lazy[node] += val
        return
    push_down(node, l, r, tree, lazy)
    mid = (l + r) // 2
    if ql <= mid: update(node * 2, l, mid, ql, qr, val, tree, lazy)
    if qr > mid: update(node * 2 + 1, mid + 1, r, ql, qr, val, tree, lazy)
    tree[node] = tree[node * 2] + tree[node * 2 + 1]

def query(node, l, r, ql, qr, tree, lazy):
    if ql <= l and r <= qr:
        return tree[node]
    push_down(node, l, r, tree, lazy)
    mid = (l + r) // 2
    res = 0
    if ql <= mid: res += query(node * 2, l, mid, ql, qr, tree, lazy)
    if qr > mid: res += query(node * 2 + 1, mid + 1, r, ql, qr, tree, lazy)
    return res
```

## 斜率优化/单调队列维护上凸壳
```Python
# 【模板积木 1】斜率优化/单调队列维护上凸壳 (求最大值)
def build_convex_hull(lines):
    """
    lines: 列表，元素为 (slope, intercept)，需按 slope 从小到大排序
    返回: 维护好的上凸壳直线序列
    """
    hull = []
    for k, b in lines:
        while len(hull) >= 2:
            k2, b2 = hull[-1]
            k1, b1 = hull[-2]
            # 交叉相乘避免浮点误差，判断中间的直线是否被“夹死”（冗余）
            if (b1 - b2) * (k - k2) >= (b2 - b) * (k2 - k1):
                hull.pop()
            else:
                break
        hull.append((k, b))
    return hull

# 【模板积木 2】凸壳二分查询
def query_max(hull, x):
    """
    在凸壳上二分查找使得 kx + b 最大的直线
    """
    l, r = 0, len(hull) - 1
    while l < r:
        mid = (l + r) // 2
        k1, b1 = hull[mid]
        k2, b2 = hull[mid + 1]
        # 如果 mid+1 的线在 x 处更优，说明最优解在右侧
        if k1 * x + b1 <= k2 * x + b2:
            l = mid + 1
        else:
            r = mid
    return hull[l][0] * x + hull[l][1]
```

```Python
# 0-1 背包 (倒序遍历)
def knapsack_01(W, w, v, n):
    dp = [0] * (W + 1)
    for i in range(1, n + 1):
        for j in range(W, w[i] - 1, -1):
            dp[j] = max(dp[j], dp[j - w[i]] + v[i])
    return dp[W]

# 完全背包 (正序遍历)
def knapsack_complete(W, w, v, n):
    dp = [0] * (W + 1)
    for i in range(1, n + 1):
        for j in range(w[i], W + 1):
            dp[j] = max(dp[j], dp[j - w[i]] + v[i])
    return dp[W]
```
